import { Component, Input } from '@angular/core';
import { ControlAccessor } from '@main/classes/control-accessor.class';

@Component({
  selector: 'app-textarea',
  templateUrl: './textarea.component.html',
  styleUrls: ['./textarea.component.scss'],
})
export class TextareaComponent extends ControlAccessor {
  /**
   * Floating label text to display
   */
  @Input() floatingLabel?: string;

  /**
   * Static label text to display
   */
  @Input() staticLabel?: string;

  /**
   * Input placeholder text
   */
  @Input() placeholder: string = '';

  /**
   * Hint to display beneath the input to provide additional information of how to use the input
   */
  @Input() hint?: string;

  @Input() rows?: number = 4;

  @Input() cols?: number = 50;
}

<div class="form-field">
  <label *ngIf="staticLabel" [class.required]="required">{{ staticLabel }}</label>
  <mat-form-field appearance="outline" [class.without-floating-label]="!floatingLabel">
    <mat-label *ngIf="floatingLabel">{{ floatingLabel }}</mat-label>
    <textarea matInput [placeholder]="placeholder" [required]="required" [formControl]="control"
      [autocomplete]="autocomplete" [rows]="rows" [cols]="cols" name="test"></textarea>
    <mat-error>{{ ngControl.control?.errors | validationError }}</mat-error>
    <mat-hint *ngIf="hint">{{ hint }}</mat-hint>
  </mat-form-field>
</div>
