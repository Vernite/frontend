import { Canvas, Source, Meta } from '@storybook/addon-docs';
// import '../../styles/themes/documentation.theme.scss';

<Meta title="Styles/Typography" />

export const Init = () => {
  document.body.style.setProperty('--color-text', '#000');
  return <div></div>
}

<Init />

export const Typeset = ({ fontWeight }) => (
  <Canvas withSource='none'>
    <div style={{ width: '100%', height: '100%' }}>
      {
        Object.entries(typography.size).map(([key, value]) => (
          <div key={key} style={{ display: 'flex', width: '100%', alignItems: 'baseline', marginBottom: '1rem' }}>
            <div style={{ width: '100px', flexShrink: 0 }}>
              <div style={{ fontSize: '14px', opacity: '0.75' }}>
                {key}
              </div>
            </div>
            <div>
              <div className={key} style={{ fontSize: `${value}px`, fontWeight: fontWeight }}>
                {SampleText}
              </div>
            </div>
          </div>
        ))
      }
    </div>
  </Canvas>
)

export const typography = {
  type: {
    primary: 'montserrat, roboto, \'Helvetica Neue\', sans-serif',
  },
  weight: {
    regular: '400',
    bold: '700',
    extrabold: '800',
    black: '900',
  },
  size: {
    'text-label': 12,
    'text-title': 14,
    'text-base': 14,
    'text-lg': 16,
    'text-xl': 18,
    'text-2xl': 20,
    'text-3xl': 24,
    'text-4xl': 28,
    'text-5xl': 32,
  },
};

export const SampleText = 'Lorem ipsum dolor sit amet, consectetur adipiscing elit.';

# Typography

## Font family

Montserrat

## Font weights

400 (`font-normal`),

700 (`font-bold`),

800 (`font-extrabold`),

900 (`font-black`)

## Classes

### Regular typeset

<Typeset fontWeight={typography.weight.regular} />

#### Example usage:
<Canvas withSource="open" withToolbar>
  <span className="text-label">Sample text</span>
  <span className="text-title">Sample text</span>
  <span className="text-base">Sample text</span>
  <span className="text-lg">Sample text</span>
  <span className="text-xl">Sample text</span>
</Canvas>

<Source
  language='html'
  dark
  format={false}
  code={`
<span class="text-label">Sample text</span>
<span class="text-title">Sample text</span>
<span class="text-base">Sample text</span>
<span class="text-lg">Sample text</span>
<span class="text-xl">Sample text</span>
  `}
/>

You can also use it directly in the `scss` files:

<Source
  language='css'
  dark
  format={false}
  code={`
selector {
  @include text-label;
}
  `}
/>
